```python
from __future__ import annotations

from dataclasses import dataclass
import json
from pathlib import Path
from typing import Any


class DatasetValidationError(ValueError):
    """Raised when a JSONL evaluation dataset is invalid."""


@dataclass(frozen=True)
class DatasetCase:
    id: str | None
    input: str
    expected_output: str | None
    retrieval_context: list[str] | None
    tags: list[str]
    metadata: dict[str, Any]
    actual_output: str | None


def load_jsonl_dataset(path: str | Path) -> list[DatasetCase]:
    dataset_path = Path(path)
    cases: list[DatasetCase] = []

    with dataset_path.open("r", encoding="utf-8") as file:
        for line_number, raw_line in enumerate(file, start=1):
            line = raw_line.strip()
            if not line:
                continue
            try:
                record = json.loads(line)
            except json.JSONDecodeError as exc:
                raise DatasetValidationError(f"line {line_number}: invalid JSON: {exc.msg}") from exc
            if not isinstance(record, dict):
                raise DatasetValidationError(f"line {line_number}: record must be an object")
            cases.append(_parse_case(record, line_number))

    if not cases:
        raise DatasetValidationError(f"{dataset_path}: dataset is empty")

    return cases


def _parse_case(record: dict[str, Any], line_number: int) -> DatasetCase:
    input_value = record.get("input")
    if not isinstance(input_value, str) or not input_value.strip():
        raise DatasetValidationError(f"line {line_number}: input must be a non-empty string")

    expected_output = record.get("expected_output")
    if expected_output is not None and not isinstance(expected_output, str):
        raise DatasetValidationError(f"line {line_number}: expected_output must be a string")

    actual_output = record.get("actual_output")
    if actual_output is not None and not isinstance(actual_output, str):
        raise DatasetValidationError(f"line {line_number}: actual_output must be a string")

    retrieval_context = record.get("retrieval_context")
    if retrieval_context is not None:
        if not isinstance(retrieval_context, list) or not all(isinstance(item, str) for item in retrieval_context):
            raise DatasetValidationError(f"line {line_number}: retrieval_context must be a list of strings")

    tags = record.get("tags", [])
    if not isinstance(tags, list) or not all(isinstance(item, str) for item in tags):
        raise DatasetValidationError(f"line {line_number}: tags must be a list of strings")

    metadata = record.get("metadata", {})
    if not isinstance(metadata, dict):
        raise DatasetValidationError(f"line {line_number}: metadata must be an object")

    case_id = record.get("id")
    if case_id is not None and not isinstance(case_id, str):
        raise DatasetValidationError(f"line {line_number}: id must be a string")

    return DatasetCase(
        id=case_id,
        input=input_value,
        expected_output=expected_output,
        retrieval_context=retrieval_context,
        tags=tags,
        metadata=metadata,
        actual_output=actual_output,
    )
```

```python
from __future__ import annotations

import os
from typing import Protocol

from dotenv import load_dotenv
from openai import OpenAI


class ChatProvider(Protocol):
    def complete(self, prompt: str) -> str:
        """Return a text completion for a user prompt."""


class OpenAIChatProvider:
    def __init__(self, model: str | None = None) -> None:
        load_dotenv()
        self._model = model or os.getenv("OPENAI_MODEL", "gpt-4.1-mini")
        self._client = OpenAI()

    def complete(self, prompt: str) -> str:
        response = self._client.responses.create(
            model=self._model,
            input=prompt,
        )
        return response.output_text.strip()
```

```python
    if app_name == "qa":
        return QAAppRunner(app=QAApp(provider=OpenAIChatProvider()))
```

```python
from __future__ import annotations

from dataclasses import asdict, dataclass, field
from datetime import UTC, datetime
import json
from pathlib import Path


@dataclass(frozen=True)
class MetricResult:
    name: str
    score: float | None
    passed: bool | None
    reason: str | None


@dataclass(frozen=True)
class ReportCase:
    id: str | None
    input: str
    actual_output: str
    metrics: list[MetricResult]


@dataclass(frozen=True)
class EvaluationReport:
    dataset_path: str
    app_name: str
    metric_preset: str
    cases: list[ReportCase]
    run_timestamp: str = field(default_factory=lambda: datetime.now(UTC).isoformat())

    def to_dict(self) -> dict[str, object]:
        payload = asdict(self)
        payload["summary"] = {
            "case_count": len(self.cases),
            "metric_averages": _metric_averages(self.cases),
        }
        return payload


def write_report(report: EvaluationReport, output_dir: str | Path = "reports") -> Path:
    directory = Path(output_dir)
    directory.mkdir(parents=True, exist_ok=True)
    timestamp = datetime.now(UTC).strftime("%Y%m%dT%H%M%SZ")
    path = directory / f"deepeval-report-{timestamp}.json"
    path.write_text(json.dumps(report.to_dict(), ensure_ascii=False, indent=2), encoding="utf-8")
    return path


def print_summary(report: EvaluationReport, report_path: Path) -> None:
    summary = report.to_dict()["summary"]
    print(f"Report: {report_path}")
    print(f"Cases: {summary['case_count']}")
    for metric_name, average in summary["metric_averages"].items():
        print(f"{metric_name}: average={average:.3f}")


def _metric_averages(cases: list[ReportCase]) -> dict[str, float]:
    values: dict[str, list[float]] = {}
    for case in cases:
        for metric in case.metrics:
            if metric.score is not None:
                values.setdefault(metric.name, []).append(metric.score)
    return {
        metric_name: sum(scores) / len(scores)
        for metric_name, scores in values.items()
        if scores
    }
```

```python
from __future__ import annotations

import os

from dotenv import load_dotenv
from deepeval.metrics import AnswerRelevancyMetric, ContextualRelevancyMetric, FaithfulnessMetric, GEval
from deepeval.test_case import SingleTurnParams


def build_metrics(preset: str):
    load_dotenv()
    model = os.getenv("DEEPEVAL_MODEL", os.getenv("OPENAI_MODEL", "gpt-4.1-mini"))

    if preset == "qa":
        return [
            AnswerRelevancyMetric(threshold=0.7, model=model, include_reason=True),
            GEval(
                name="Correctness",
                evaluation_steps=[
                    "Compare the actual output with the expected output.",
                    "Penalize factual contradictions.",
                    "Penalize missing details that are necessary to answer the input.",
                    "Accept wording differences when the meaning is equivalent.",
                ],
                evaluation_params=[
                    SingleTurnParams.INPUT,
                    SingleTurnParams.ACTUAL_OUTPUT,
                    SingleTurnParams.EXPECTED_OUTPUT,
                ],
                threshold=0.7,
                model=model,
            ),
        ]

    if preset == "rag":
        return [
            AnswerRelevancyMetric(threshold=0.7, model=model, include_reason=True),
            FaithfulnessMetric(threshold=0.7, model=model, include_reason=True),
            ContextualRelevancyMetric(threshold=0.7, model=model, include_reason=True),
        ]

    raise ValueError("unknown metric preset: " + preset + ". Supported presets: qa, rag")
```

```python
from __future__ import annotations

from pathlib import Path

from deepeval.test_case import LLMTestCase

from llm_eval.app_runner import AppRunner
from llm_eval.datasets import DatasetCase, DatasetValidationError, load_jsonl_dataset
from llm_eval.metrics import build_metrics
from llm_eval.reports import EvaluationReport, MetricResult, ReportCase


def run_evaluation(
    dataset_path: str | Path,
    app_name: str,
    app_runner: AppRunner,
    metric_preset: str,
) -> EvaluationReport:
    cases = load_jsonl_dataset(dataset_path)
    _validate_cases_for_preset(cases, metric_preset)
    metrics = build_metrics(metric_preset)

    report_cases: list[ReportCase] = []
    for case in cases:
        app_result = app_runner.run(case.input)
        test_case = LLMTestCase(
            input=case.input,
            actual_output=app_result.actual_output,
            expected_output=case.expected_output,
            retrieval_context=app_result.retrieval_context or case.retrieval_context,
        )
        metric_results = []
        for metric in metrics:
            metric.measure(test_case)
            metric_results.append(
                MetricResult(
                    name=getattr(metric, "name", metric.__class__.__name__),
                    score=getattr(metric, "score", None),
                    passed=getattr(metric, "success", None),
                    reason=getattr(metric, "reason", None),
                )
            )
        report_cases.append(
            ReportCase(
                id=case.id,
                input=case.input,
                actual_output=app_result.actual_output,
                metrics=metric_results,
            )
        )

    return EvaluationReport(
        dataset_path=str(dataset_path),
        app_name=app_name,
        metric_preset=metric_preset,
        cases=report_cases,
    )


def _validate_cases_for_preset(cases: list[DatasetCase], metric_preset: str) -> None:
    if metric_preset == "qa":
        for index, case in enumerate(cases, start=1):
            if not case.expected_output:
                case_name = case.id or f"line {index}"
                raise DatasetValidationError(f"{case_name}: expected_output is required for qa preset")
        return

    if metric_preset == "rag":
        for index, case in enumerate(cases, start=1):
            if not case.retrieval_context:
                case_name = case.id or f"line {index}"
                raise DatasetValidationError(f"{case_name}: retrieval_context is required for rag preset")
        return

    raise ValueError("unknown metric preset: " + metric_preset + ". Supported presets: qa, rag")
```

```python
from __future__ import annotations

import argparse

from llm_eval.app_runner import build_app_runner
from llm_eval.deepeval_runner import run_evaluation
from llm_eval.reports import print_summary, write_report


def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(prog="llm-eval")
    subparsers = parser.add_subparsers(dest="command", required=True)

    run_parser = subparsers.add_parser("run", help="Run local DeepEval evaluation")
    run_parser.add_argument("--dataset", required=True)
    run_parser.add_argument("--app", default="qa", choices=["qa"])
    run_parser.add_argument("--preset", default="qa", choices=["qa", "rag"])
    run_parser.add_argument("--output-dir", default="reports")

    return parser


def main() -> None:
    args = build_parser().parse_args()

    if args.command == "run":
        app_runner = build_app_runner(args.app)
        report = run_evaluation(
            dataset_path=args.dataset,
            app_name=args.app,
            app_runner=app_runner,
            metric_preset=args.preset,
        )
        report_path = write_report(report, output_dir=args.output_dir)
        print_summary(report, report_path)
```
